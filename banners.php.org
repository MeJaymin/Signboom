  function DisplayProductDescriptions() {
    global $database_DBConn, $DBConn, $first;

    $Qry = "SELECT * FROM signboom_allproducts WHERE Category = 'BANNER' AND Enabled = 1 ORDER BY SortGroup, SortOrder";

       $first = false;
       echo '<br>';
     }

  }
